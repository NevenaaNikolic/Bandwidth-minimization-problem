Funkcija oznaavanja $f:\{v_1,v_2,...,v_n\} \rightarrow \{1,2,...,n\}$.

Definiemo dve pomone funkcije. $labeling\_from\_nodes$ je pomona funkcija koja nam od liste vorova $v_j$ daje listu indeksa $j$ (tj. oznaavanje tih vorova). $nodes\_from\_labeling$ je pomona funkcija koja nam od liste indeksa $j$ daje listu vorova $v_j$ (tj. vorove na osnovu tog labeliranja).

In [195]:
def labeling_from_nodes(nodes):
    n = len(nodes)
    return [int(nodes[i].strip('v')) for i in range(n)]

def nodes_from_labeling(labeling):
    n = len(labeling)
    return ['v'+str(labeling[i]) for i in range(n)]

Definiemo funkciju $bandwidth\_of\_a\_node$ koja za dati graf $G$, vor $node$ i funkciju oznaavanja $f$ izraunava $bandwidth$ tog vora

In [199]:
def bandwidth_of_a_node(G, node, f):
    neighbors = [u for u in G.neighbors(node)]
    label_node = label(G,node,f)
    maximum = float('-inf')
    for u in neighbors:
        label_u = label(G,u,f)
        diff = abs(label_u - label_node)
        if diff > maximum:
            maximum = diff
    return maximum

Definiemo funkciju $bandwidth$ koja za dati graf $G$ i funkciju oznaavanja $f$ izraunava $bandwidth$ grafa

In [201]:
def bandwidth(G, f):
    nodes = [v for v in G.nodes]
    bandwidths = [bandwidth_of_a_node(G,node,f) for node in nodes]
    return max(bandwidths)

Definiemo funkciju $initial\_solution$ koja za dati graf $G$ generie poetno reenje - poetno oznaavanje $f$ vorova grafa $G$, koristei BFS sa nasuminim izborom vora od kojeg se zapoinje pretraga.

In [203]:
def initial_solution(G):
    
    solution = []
    
    for C in nx.connected_components(G):
        source = random.choice(list(C))
        edges_iterator = nx.bfs_edges(G, source)
        edges = [edge for edge in edges_iterator]
        solution = solution + [source] + [edge[1] for edge in edges]
    
    
    return labeling_from_nodes(solution)

In [205]:
def distance(f,f_p):
    n = len(f)
    dist = 0
    for i in range(n):
        if f[i] != f_p[i]:
            dist += 1
    dist -= 1
    return dist

Definiemo funkciju $set\_K$ koja bira podskup vorova $K$ na osnovu $Initialization$ koraka u $Algorithm 1$ iz rada.

In [210]:
def set_K(G, f, k):
    nodes = [u for u in G.nodes]
    B = bandwidth(G,f)
    B_p = random.randint(1,B)
    
    K = []
    while True:
        for node in nodes:
            bandwidth_node = bandwidth_of_a_node(G, node, f)
            if bandwidth_node >= B_p:
                K.append(node)
                
        if len(K) >= k:
            break
        else:
            K = []
            B_p = random.randint(1,B)
                
    return K

Definiemo funkciju $critical\_node$ koja za dati graf $G$, vor $node$ i oznaavanje $f$ nalazi vor $v$ takav da je $ |f(node)-f(v)| = bandwidth\_of\_a\_node(G, node, f) $

In [212]:
def critical_node(G, node, f):

    nodes = [u for u in G.nodes]
    for v in nodes:
        if abs(label(G, node, f) - label(G, v, f)) == bandwidth_of_a_node(G, node, f):
            return v

Definiemo pomonu funkciju $swap\_labels$ koja menja oznaavanje $f$ tako to datim vorovima $u$ i $v$ zameni prethodne oznake.

In [214]:
def swap_labels(G, u, v, f):
    
    label_u = label(G, u, f)
    label_v = label(G, v, f)
    
    index_u = f.index(label_u)
    index_v = f.index(label_v)
    
    f[index_u] = label_v
    f[index_v] = label_u
    
    return f

Definiemo funkciju $shaking$ kojom se realizuje faza razmrdavanja (prema $Algorithm 1$ iz rada)

In [216]:
def shaking(G, f, k):
    
    K = set_K(G, f, k)
    
    edges = [edge for edge in G.edges]
    nodes = [node for node in G.nodes]
    
    for i in range(k):
        u = random.choice(K)
        v = critical_node(G, u, f) 
        
        if (u,v) in edges:
            f_min_u = min_labeled_neighbor(G, u, f)
            f_max_u = max_labeled_neighbor(G, u, f)

            label_v = label(G, v, f)
            
            min_value = float('inf')
            min_w = None
            
            for w in nodes:
                label_w = label(G, w, f)
                if label_w <= f_max_u and label_w >= f_min_u:
                    current_value = max(max_labeled_neighbor(G, w, f) - label_v,label_v - min_labeled_neighbor(G, w, f))
                    
                    if current_value < min_value:
                        min_value = current_value
                        min_w = w
                        
            f = swap_labels(G,v,min_w,f)
            
    return f

Definiemo funkciju $best\_labeling$ koja za dati vor $v$ pronalazi najbolju oznaku.

In [218]:
def best_labeling(G, v, f):
    max_v = max_labeled_neighbor(G, v, f)
    min_v = min_labeled_neighbor(G, v, f)
    
    mid_v = (max_v + min_v) // 2
    
    return mid_v

Definiemo funkciju $suitable\_swapping\_nodes$ koja za dati vor $v$ pronalazi skup vorova pogodnih za razmenu oznake.

In [219]:
def suitable_swapping_nodes(G, v, f):
    nodes = [u for u in G.nodes]
    mid_v = best_labeling(G, v, f)
    label_v = label(G, v, f)
    N_p = []
    
    for u in nodes:
        label_u = label(G, u, f)
        if abs(mid_v-label_u) < abs(mid_v-label_v):
            N_p.append(u)
            
    return N_p

Definiemo funkciju $number\_of\_critical\_edges$ koja za dati graf $G$ i oznaavanje $f$ izraunava broj kritinih grana. Kritina grana je ona grana $(u,v)$ takva da je $bandwidth\_of\_a\_node(G, u, f) = bandwidth(G, f)$ i $bandwidth\_of\_a\_node(G, v, f) = bandwidth(G, f)$

In [221]:
def number_of_critical_edges(G, f):
    
    number = 0
    B = bandwidth(G, f)
    edges = [edge for edge in G.edges]
    
    for edge in edges:
        v_from = edge[0]
        v_to = edge[1]
        
        if bandwidth_of_a_node(G, v_from, f) == B and bandwidth_of_a_node(G, v_to, f) == B:
            number += 1
            
    return number

Definiemo funkciju $local\_search$ kojom se realizuje faza lokalne pretrage (prema $Algorithm 2$ iz rada)

In [222]:
def local_search(G, f):
    
    canImprove = True
    nodes = [u for u in G.nodes]
    
    while canImprove:
        canImprove = False
        
        for v in nodes:
            
            number_of_critical = number_of_critical_edges(G, f)

            if bandwidth_of_a_node(G, v, f) == bandwidth(G, f):
                
                N_p = suitable_swapping_nodes(G,v,f)
                
                for u in N_p:
                    f = swap_labels(G,v,u,f)
                    
                    if number_of_critical_edges(G, f) < number_of_critical:
                        canImprove = True
                        break
                        
                    f = swap_labels(G,v,u,f)
                    
    return f

Definiemo funkciju $number\_of\_critical\_nodes$ koja za dati graf $G$ i oznaavanje $f$ izraunava broj kritinih vorova. Kritini vor je vor iji je $bandwidth$ jednak $bandwidt$h-u grafa, odnosno ima maksimalan $bandwidth$ za dato labeliranje.

In [223]:
def number_of_critical_nodes(G, f):
    
    Vc = set([])
    B = bandwidth(G, f)
    nodes = [u for u in G.nodes]
    
    for u in nodes:
        if bandwidth_of_a_node(G, u, f) == B:
            Vc.add(u)
            
    return len(Vc)

Definiemo funkciju $move$ kojom se realizuje mehanizam 'move or not' (prema $Algorithm 3$ iz rada)

In [224]:
def move(G, f, f_p, alpha):
    
    Move = False
    B_f = bandwidth(G, f)
    B_f_p = bandwidth(G, f_p)
    
    if B_f_p < B_f:
        Move = True
        
    else:
        if B_f == B_f_p:
            if number_of_critical_nodes(G,f_p) < number_of_critical_nodes(G,f) or distance(f, f_p) > alpha:
                Move = True
                
    return Move

Definiemo funkciju $VNS$ kojom se realizuje metod promenljivih okolina (prema $Algorithm 4$ iz rada)

def VNS(G, k_min, k_max, k_step, alpha):

    t = 0
    
    i_max = int((k_max - k_min)/k_step)
    
    f = initial_solution(G)
    f = local_search(G, f)
    i = 0
    k = k_min
    
    while i <= i_max:
        f_p = shaking(G, f, k)
        f_p = local_search(G, f)
        if move(G, f, f_p, alpha):
            f = f_p
            k = k_min
            i = 0
        else:
            k = k + k_step
            i = i + 1
    
    return f

Pomona funkcija $generate\_graph$ koja na osnovu broja vorova $n$ i broja grana $e$ vraa nasumino konstruisan graf.

In [226]:
def generate_graph(n, e):
    
    l = [i for i in range(1,n+1)]
    V = nodes_from_labeling(l)
    G = nx.Graph()
    G.add_nodes_from(V)

    for i in range(e):
        u = random.choice(V)
        v = random.choice(V)
    
        while u == v:
            v = random.choice(V)
        
        G.add_edge(u,v)
        
    return G
